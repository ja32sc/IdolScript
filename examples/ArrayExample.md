idol unit = [];  
unit.addMember("Ai");  
unit.addMember("Nino")

perform(unit);  // Output: [ "Ai", "Nino" ]; 

// Graduating a member (removing them from the group)
idol graduatedMember = unit.graduate();  
perform(graduatedMember); 

perform(unit);  // Output: [ Ai ]
